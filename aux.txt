void update_arp_entry(struct arp_table_entry *arp_table, int arp_table_len, struct arp_table_entry new_entry)
{
	for (int i = 0; i < arp_table_len; i++)
	{
		if (arp_table[i].ip == new_entry.ip)
		{
			memcpy(arp_table[i].mac, new_entry.mac, 6);
			return;
		}
	}
	arp_table[arp_table_len] = new_entry;
	arp_table_len++;
}
